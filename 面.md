3. 连等符号
- [详细](http://www.iteye.com/topic/785445)
```
// a
var a = {n:1};  
a.x = a = {n:2};  
alert(a.x);  // --> undefined 

// b
var a = {n:1};  
var b = a;  // 持有a，以回查  
a.x = a = {n:2};  
alert(a.x);   // --> undefined  
alert(b.x);   // --> [object Object]  

// c
function fun(){  
    var a = b = 5;  
}  
fun();  
alert(typeof a);   // --> undefined  
alert(typeof b);   // --> number
```
